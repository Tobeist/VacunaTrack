#KPIFórmula / FuenteSignificado operativo
1Total pacientesCOUNT(*) FROM vw_pacientesNiños/as registrados en el sistema
2Total tutoresCOUNT(*) FROM vw_tutoresTutores con acceso al portal
3Total responsablesCOUNT(*) FROM vw_responsablesPersonal clínico con acceso
4Total centrosCOUNT(*) FROM vw_centros_detalleCentros de salud registrados
5Centros activos (30d)COUNT(DISTINCT centro_id) WHERE ts ≥ NOW()-30dCentros con al menos una aplicación en los últimos 30 días
6% Centros activos (30d)centros_activos_30d / total_centros × 100Proporción de la red que estuvo operativa recientemente
7Total aplicacionesCOUNT(*) FROM aplicacionesDosis históricamente administradas
8Aplicaciones hoyCOUNT(*) WHERE DATE(timestamp) = CURRENT_DATEDosis aplicadas en el día actual
9Aplicaciones este mesCOUNT(*) WHERE DATE_TRUNC('month') = MES_ACTUALRitmo de vacunación mensual
10Promedio diario (mes)aplicaciones_mes / EXTRACT(DAY FROM NOW())Cadencia diaria de vacunación en el mes en curso
11Cobertura globalCOUNT(DISTINCT paciente_id) / total_pacientes × 100% de pacientes con al menos una vacuna aplicada
12Pacientes sin vacunarCOUNT(*) WHERE NOT EXISTS aplicacionesPacientes registrados que aún no tienen ninguna dosis
13Vacunas en catálogoCOUNT(*) FROM vacunasVacunas disponibles en el programa
14Esquemas de vacunaciónCOUNT(*) FROM esquemasEsquemas oficiales registrados (p. ej. infantil, adulto)
15Padecimientos cubiertosCOUNT(*) FROM padecimientosEnfermedades que el programa previene
16Lotes activos con stockCOUNT(*) FROM vw_inventarios WHERE stock > 0 AND fecha_cad ≥ HOYLotes vigentes con dosis disponibles
17Lotes por caducar (30d)COUNT(*) FROM vw_inventarios WHERE fecha_cad BETWEEN HOY AND HOY+30Lotes que requieren atención inmediata por proximidad de caducidad
18Lotes caducados con stockCOUNT(*) FROM vw_inventarios WHERE fecha_cad < HOY AND stock > 0Lotes vencidos que aún tienen dosis — deben retirarse
19Alertas de inventarioCOUNT(*) FROM vw_alertas_inventarioAlertas activas de stock bajo o caducidad
20Alertas de dosisCOUNT(*) FROM vw_alertas_dosisPacientes con dosis atrasadas o próximas a vencer el límite de edad
